## 使用指南

```tsx
import React, { useState } from 'react'
import { View } from '@tarojs/components'
import { HuiSwitch } from '@zaihui/hui-design'

const Demo: React.FC = () => {
  const [checked, setChecked] = useState(false)
  return (
    <View>
      <Switch checked={checked} onChange={(v) => setChecked(v)} />
    </View>
  )
}

export default Demo
```
